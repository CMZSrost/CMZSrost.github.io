设计方法

读取单词表

出于时间复杂度优化的角度，采用字符树的方式进行单词纠错，故此处用构建字符树的方法来读取单词表（具体实现在算法分析叙述，此处不再赘述，下同）。

首先确定字符树结构：使用散列表映射对应字符，在将单词储存在字符树叶节点（相对） 


/*字符树
alpha, 字符串
alpha_tree, 下个节点
*/
typedef struct alpha_tree
{
    char* alpha;
    struct alpha_tree *next[alphas];
}alphaT;

同时构建（逆）哈希转换函数：0 ~ 25为小写字母a ~ z，26 ~ 51表示大写字母A ~ Z


char chart(int i){
    if(i<26)return (char)(i+'a');
    else return (char)(i-26+'A');
}

int rechart(char i){
    return
    (int)i-'a'>=0?
    (int)(i - 'a') :
    (int)(i - 'A' + 26);
}

接着构建字符树，在构建前，使用函数初始化节点


/*新增树节点
--  input:
    void
--  output
    alphaT* 实例化节点
*/
alphaT *init()
{
    alphaT *target = (alphaT *)malloc(sizeof(alphaT));
    target->alpha = NULL;
    for (int i = 0; i < alphas; i++)
        target->next[i] = NULL;
    return target;
};

/*构造树
--  input
    int 当前遍历位置
    char* 字符串
    int 字符长度
    alphaT* 当前指向节点
--  output
    void
*/
void set_tree(int x, char *str, int slen, alphaT *cur)
{
    if (x==slen)
    {
        cur->alpha = str;
        cur->alpha[slen]='\0';
        return;
    }
    int store = rechart(str[x]);
    if (NULL == cur->next[store])
    {
        cur->next[store] = init();
    }
    set_tree(x + 1, str, slen, cur->next[store]);
    return;
};

/*释放树空间
--  input

void free_tree(alphaT *target)
{
    for (int i = 0; i < alphas; i++)
    {
        if (target->next[i])
            free_tree(target->next[i]);
    }
    if(target->alpha)
        free(target->alpha);
    free(target);
    return;
};

void cmpword(char *line, int len)
{
    char str_tmp[MAX_LINE_LEN];
    int str_len = 0;
    int cur = 0;

    while (!isalpha(line[cur]))
        cur++;
    while (isalpha(line[cur]))
    {
        str_tmp[str_len] = line[cur];
        str_len++;
        cur++;
    }
    str_tmp[str_len] = '\0';

    putstr(str_tmp, str_len);
    str_len = 0;

    // printf("%s ",tmp);
    fputc(' ', correct);
    while (cur < len && line[cur] == ' ')
    {
        cur++;
    }

    while (cur < len)
    {
        if (isalpha(line[cur]))
        {
            str_tmp[str_len++] = line[cur++];
        }
        else if (!isalpha(line[cur]))
        {

            putstr(str_tmp, str_len);

                cur++;

                fputc('\n', correct);
        }
    }
    return;

在单词比较时，涉及到find(),reset()函数将在下一步叙述

单词比较

单词比较是这个程序的核心算法，对其的优化程度决定了整个程序的运行效率，为了优化，将设计两个函数find()（剪枝操作）、edit_dis()（比较编辑距离）（具体实现在算法分析叙述，此处不赘述）

首先设计find()函数在字符树中寻找最有可能的单词，并且在每次搜索操作中更新边界，缩小范围以优化时间复杂度


/*搜索可能单词
--  output:
    char* 待检查单词
    int 单词长度
    int 当前已检查位置
    alphaT* 当前所在节点
--  output:
    void
*/
void find(char *word, int len, int x, alphaT *cur)
{
    int now = fabs(len - x - 1);
    if (x == len && cur->alpha && !strcmp(cur->alpha, word))
    {
        res.alpha = cur->alpha;
        res.edit_dis = 0;
        return;
    }
    else if (x >= len)
    {
        if (now >= res.edit_dis)
            return;
        else
        {
            if (cur->alpha)
                editDis(word, cur->alpha);
            for (int i = 0; i < alphas; i++)
            {
                if (cur->next[i])
                {
                    find(word, len, x + 1, cur->next[i]);
                    if (now >= res.edit_dis)
                        return;
                }
            }
        }
    }
    else
    {
        int ptr = rechart(word[x]) % alphas;
        for (int i = 0; i < alphas; i++, ptr = (ptr + 1) % alphas)
        {
            if (cur->next[ptr])
            {
                find(word, len, x + 1, cur->next[ptr]);
                if (now > res.edit_dis)
                    return;
            }
        }
        if (now >= res.edit_dis)
            return;
        if (cur->alpha)
            editDis(word, cur->alpha);
        return;
    }
};

完成上述两步后，把待纠错单词与符合搜索范围的单词进行比较，由于比较中需要确定较小值，故设计min()，函数不赘述


/*计算编辑距离
--  input:
    char* 检查单词
    char* 字典单词
--  output:
    void
*/
void editDis(char *word, char *voc)
{
    int wlen = strlen(word) + 1, vlen = strlen(voc) + 1;
    int **dp;
    dp = (int **)calloc(sizeof(int *), wlen);
    for (int i = 0; i < wlen; i++)
    {
        dp[i] = (int *)calloc(sizeof(int), vlen);
        dp[i][0] = i;
    }
    for (int i = 0; i < vlen; i++)
        dp[0][i] = i;

    for (int i = 1; i < wlen; i++)
    {
        for (int j = 1; j < vlen; j++)
        {
            if (word[i] == voc[j])
                dp[i][j] = min(
                    min(dp[i - 1][j] + 1,
                        dp[i][j - 1] + 1),
                    dp[i - 1][j - 1]);
            else
                dp[i][j] = 1 + min(
                                   min(dp[i - 1][j],
                                       dp[i][j - 1]),
                                   dp[i - 1][j - 1]);
        }
    }
    if (dp[wlen - 1][vlen - 1] < res.edit_dis)
    {
        res.alpha = voc;
        res.edit_dis = dp[wlen - 1][vlen - 1];
    }
    for (int i = 0; i < wlen; i++)
    {
        if (dp[i] != NULL)
            free(dp[i]);
    }
    free(dp);
    return;
}

算法分析

计算枚举法可行性

对于文档提供的编辑距离，采用了动态规划的思想，单纯来比较某一单词

(设单个单词长度为M，用来比较的单词长度为N)

时间复杂度为：O(MN)，空间复杂度为O(MN)

同时考虑到每个单词都进行对整个单词表的遍历，可知

(设待纠错单词数为A，单词表单词数为B)

时间复杂度为O(ABMN),空间复杂度为O(MN+AB)

显然靠枚举法进行比较，程序的时间开销极大，不予考虑

算法改良-剪枝

注意到时间复杂度主要是由于可比较单词过多导致的，故考虑用剪枝的方法来缩小搜索范围，提高效率

关注程序目的可知，只需构建有利于字符匹配的结构体，通过哈希映射的方式，便可将最优时间复杂度降到O(N)，所需付出的空间复杂度代价仅为O(BN)。

故使用字符树结构，以ascii码作为键值偏移量，构建了字符的哈希表
由于字符树到达叶节点的路径是字符串的映射，故可仅在结点处赋值有效字符串（有效指单词表中存在该单词）来表示该处为一“叶”节点（代表字符串路径的结束）

完成结构上的剪枝后，接下来是搜索范围的剪枝

对于非最优情况，可以以最优情况为起点考虑问题。观察可知编辑距离总大于等于两字符串的长度之差，由此可以使搜索优先从同等长度开始，向更长的字符串遍历；与此同时，利用动态规划的思维，每进行一次搜索决策时，参考当前最优编辑距离，缩小搜索范围。

剪枝环节用递归函数find来实现

剪枝实现－find()

首先，设置游标x与cur表示当前函数检查的字符位置，len表示检查的单词的长度，word传递待检查字符串


/*搜索可能单词
--  output:
    char* 待检查单词
    int 单词长度
    int 当前已检查位置
    alphaT* 当前所在节点
--  output:
    void
*/
void find(char *word, int len, int x, alphaT *cur)
{
    int now = fabs(len - x - 1);
    if (x == len && cur->alpha && !strcmp(cur->alpha, word))
    {
        res.alpha = cur->alpha;
        res.edit_dis = 0;
        return;
    }
    else if (x >= len)
    {
        if (now >= res.edit_dis)
            return;
        else
        {
            if (cur->alpha)
                editDis(word, cur->alpha);
            for (int i = 0; i < alphas; i++)
            {
                if (cur->next[i])
                {
                    find(word, len, x + 1, cur->next[i]);
                    if (now >= res.edit_dis)
                        return;
                }
            }
        }
    }
    else
    {
        int ptr = rechart(word[x]) % alphas;
        for (int i = 0; i < alphas; i++, ptr = (ptr + 1) % alphas)
        {
            if (cur->next[ptr])
            {
                find(word, len, x + 1, cur->next[ptr]);
                if (now > res.edit_dis)
                    return;
            }
        }
        if (now >= res.edit_dis)
            return;
        if (cur->alpha)
            editDis(word, cur->alpha);
        return;
    }
};

由于字符树”叶”结点并不代表枝干的结束，因此此处x与len共有三种不同意义的大小关系x==len,x>=len与x < len

首先讨论x==len，此时字符路径与word长度相等，若该结点为”叶”结点，表示该word有对应的单词，故可赋值到当前curpack并返回；若该结点不是”叶”结点，说明无对应单词，此处便需开始搜索。

当x>=len时，必定处于搜索状态，此时考虑剪枝。

定义now为x与len的绝对值距离，显然有编辑距离>=now。故当最优编辑距离<=now时，不必搜索；否则按字符树起始点开始遍历，同时运用”运行－决策”思想，每一次遍历都进行判断，及时结束循环。同时由于字符树构建过程也是剪枝，也优化了时间复杂度。

当x < len时，有可能处于搜索状态，故在循环体中加入判断，同时为了防止判断错误，需先将最优编辑距离初始化为极大值，防止还未搜索便退出，同时此处的判断也可作为函数返回结果的通道

同时，当循环体结束时，判断当前now是否大于最优编辑距离以剪枝（同时也是返回结果的通道）；若否，且该结点为”叶”结点，则进行单词比较，更新最优编辑距离

剪枝总体结构叙述结束，同时注意到一些细节：在x < len的循环体中，now与最优编辑距离使用>判断，而在其他位置则用>=判断，这是由于此处为从父向子结点遍历，若用>=，会剔除父结点，导致剪去该父结点下的可能存在的更优编辑距离，其他位置则无此问题。

编辑距离算法

最后是主要算法edit_dis()。运用动态规划方法得到编辑距离。

动态规划: 求解 以时间划分阶段的动态过程最优化方案 的过程，是一种 通过储存前阶段的结果来辅助现阶段决策 的算法

(Levenshtein)莱文斯坦距离:是编辑距离的一种。指两个字串之间，由一个转成另一个所需的最少编辑操作次数。允许的编辑操作包括将一个字符替换成另一个字符，插入一个字符，删除一个字符。

以下是pdf中对于(Levenshtein)莱文斯坦距离的解释

首先，假设对于字符串A和B，我们使用一个二维数组distance[1 + length of A][1+length of B]，

其中distance[i][j]的值表示将 A的1到i部分 转换为 B的1到j部分 需要的最少编辑次数。所以，在i等于0时，也就是说对于distance[0,j]，A的空子串部分到B的前j个元素组成的子串的转换，需要j次操作。同理可得distance数组的首行和首列为0到i和0到j。

对于剩余的distance[i][j]（也就是A1i元素的子串和B1j元素的子串的距离），可以由distance[i-1][j]，distance[i][j-1]，distance[i-1][j-1]来确定，

如果计算A[i]和B[j]相同，取min（distance[i-1][j]+1， distance[i][j-1]+1，distance[i-1][j-1]）即可。

同理，如果计算A[i]和B[j]不同，就要取min（distance[i-1][j]+1，distance[i][j-1]+1，distance[i-1][j-1]+1）。

计算出所有distance元素后，我们要求的A到B的编辑距离就是distance[1+length of A][1+length of B]

表格如下
$$
\begin{matrix}
· & · & u & c & k \\
· & 0 & 1 & 2 & 3 \\
l & 1 & 1 & 2 & 3 \\
u & 2 & 1 & 2 & 3 \\
c & 3 & 2 & 1 & 2 \\
k & 4 & 3 & 2 & 1
\end{matrix}
$$

算法复杂度计算

综上，算法的复杂度如下

设待纠错单词平均长度为A，数量为N；单词表单词平均长度为B，数量为M

假设有T个平均长度为H的错误文本

时间复杂度：O(2BM+(N-T)A+TBH(logMlogH))=O(x^3 log^2 x)

最优：Ω(2BM+(N-T)A+TBH)=Ω(x^3)

空间复杂度：O(BM+HBlogM)=O(x^2 logx)

void putstr(char *str_tmp, int str_len)
{
    char *tmp;
    tmp = (char *)calloc(sizeof(char), str_len + 1);
    strncpy(tmp, str_tmp, str_len);
    tmp[str_len] = '\0';
    find(tmp, str_len, 0, root);
    free(tmp);
    tmp = (char *)calloc(sizeof(char), strlen(res.alpha) + 1);

void cmpword(char *line, int len)
{
    char str_tmp[MAX_LINE_LEN];
    int str_len = 0;
    int cur = 0;

    while (!isalpha(line[cur]))
        cur++;
    while (isalpha(line[cur]))
    {
        str_tmp[str_len] = line[cur];
        str_len++;
        cur++;
    }
    str_tmp[str_len] = '\0';

    putstr(str_tmp, str_len);
    str_len = 0;

    // printf("%s ",tmp);
    fputc(' ', correct);
    while (cur < len && line[cur] == ' ')
    {
        cur++;
    }

    while (cur < len)
    {
        if (isalpha(line[cur]))
        {
            str_tmp[str_len++] = line[cur++];
        }
        else if (!isalpha(line[cur]))
        {

            putstr(str_tmp, str_len);
            str_len = 0;

            // printf("%s ",tmp);
            cur++;
            while (cur < len && !isalpha(line[cur]))
            {
                cur++;
            }
            if (cur != len)
                fputc('/', correct);
            else
                fputc('\n', correct);
        }
    }
    return;
};

void find(char *word, int len, int x, alphaT *cur)
{
    int now = fabs(len - x - 1);
    if (x == len && cur->alpha && !strcmp(cur->alpha, word))
    {
        res.alpha = cur->alpha;
        res.edit_dis = 0;
        return;
    }
    else if (x >= len)
    {
        if (now >= res.edit_dis)
            return;
        else
        {
            if (cur->alpha)
                editDis(word, cur->alpha);
            for (int i = 0; i < alphas; i++)
            {
                if (cur->next[i])
                {
                    find(word, len, x + 1, cur->next[i]);
                    if (now >= res.edit_dis)
                        return;
                }
            }
        }
    }
    else
    {
        int ptr = rechart(word[x]) % alphas;
        for (int i = 0; i < alphas; i++, ptr = (ptr + 1) % alphas)
        {
            if (cur->next[ptr])
            {
                find(word, len, x + 1, cur->next[ptr]);
                if (now > res.edit_dis)
                    return;
            }
        }
        if (now >= res.edit_dis)
            return;
        if (cur->alpha)
            editDis(word, cur->alpha);
        return;
    }
};

void editDis(char *word, char *voc)
{
    int wlen = strlen(word) + 1, vlen = strlen(voc) + 1;
    int **dp;
    dp = (int **)calloc(sizeof(int *), wlen);
    for (int i = 0; i < wlen; i++)
    {
        dp[i] = (int *)calloc(sizeof(int), vlen);
        dp[i][0] = i;
    }
    for (int i = 0; i < vlen; i++)
        dp[0][i] = i;

    for (int i = 1; i < wlen; i++)
    {
        for (int j = 1; j < vlen; j++)
        {
            if (word[i] == voc[j])
                dp[i][j] = min(
                    min(dp[i - 1][j] + 1,
                        dp[i][j - 1] + 1),
                    dp[i - 1][j - 1]);
            else
                dp[i][j] = 1 + min(
                                   min(dp[i - 1][j],
                                       dp[i][j - 1]),
                                   dp[i - 1][j - 1]);
        }
    }
    if (dp[wlen - 1][vlen - 1] < res.edit_dis)
    {
        res.alpha = voc;
        res.edit_dis = dp[wlen - 1][vlen - 1];
    }
    for (int i = 0; i < wlen; i++)
    {
        if (dp[i] != NULL)
            free(dp[i]);
    }
    free(dp);
    return;
}
